Squish-Skript — Echter Code (TC-1042)

# Hier verbinden wir uns zur laufenden Anwendung
# (Attachable AUT — die App läuft schon, Squish hängt sich nur dran)
def main():
    startApplication("LaserSystemUI")

    # Wir warten bis der Hauptbildschirm da ist,
    # bevor wir irgendwas anklicken
    waitForObject(":MainWindow")

    # Jetzt klicken wir auf "Neuer Patient"
    # Squish findet den Button über die Object-Map (objectName)
    clickButton(waitForObject(":btnNewPatient"))

    # Pflichtfelder ausfüllen
    type(waitForObject(":nameField"), "Mustermann, Max")
    type(waitForObject(":dobField"), "01.01.1980")
    type(waitForObject(":patientIdField"), "P-2026-1042")

    # Speichern und prüfen ob der Patient in der Liste auftaucht
    clickButton(waitForObject(":btnSave"))
    test.verify(object.exists(":patientList_Mustermann"))
